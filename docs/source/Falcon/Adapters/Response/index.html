Append a list of newline encoded headers.

def self.wrap_headers(fields)
	headers = ::Protocol::HTTP::Headers.new
	meta = {}
	
	fields.each do |key, value|
		key = key.downcase
		
		if key.start_with?('rack.')
			meta[key] = value
		else
			value.to_s.split("\n").each do |part|
				headers.add(key, part)
			end
		end
	end
	
	return headers, meta
end
